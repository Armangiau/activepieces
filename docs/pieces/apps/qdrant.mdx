## Actions

  <Card title="Add points to collection">
    Instert a point (= embedding or vector + other infos) to a specific collection, if the collection does not exist it will be created
    <Expandable title="Properties">
      <ResponseField name="Collection Name" required type="Short_text">
        The name of the collection needed for this action
      </ResponseField>
      <ResponseField name="Embeddings" required type="Short_text">
        Embeddings (= vectors) for the points
      </ResponseField>
      <ResponseField name="Embeddings Ids" required type="Short_text">
        The ids of the embeddings for the points
      </ResponseField>
      <ResponseField name="Calculation Method of distance" type="Dropdown" required>
        The calculation method helps to rank vectors when you want to find the closest points, the method to use depends on the model who's created the embeddings, see the documentation of your model
      </ResponseField>
      <ResponseField name="Payloads" type="Object">
        The additional informations for the points
      </ResponseField>
      <ResponseField name="Storage" type="Dropdown">
        Define where points will be stored
      </ResponseField>
    </Expandable>
  <Card/>

  <Card title="Search Points">
    Search for points closest to your given vector (= embedding)
    <Expandable title="Properties">
      <ResponseField name="Collection Name" required type="Short_text">
        The name of the collection needed for this action
      </ResponseField>
      <ResponseField name="Embedding" type="Short_text" required>

      </ResponseField>
      <ResponseField name="Must Have" type="Object">

      </ResponseField>
      <ResponseField name="Negative Vector" type="Short_text">

      </ResponseField>
      <ResponseField name="Limit Result" type="Number">
